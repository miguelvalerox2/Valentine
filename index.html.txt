<script>
const noBtn = document.getElementById('noBtn');
const yesBtn = document.getElementById('yesBtn');
const stage = document.getElementById('stage');
const screen1 = document.getElementById('screen1');
const screen2 = document.getElementById('screen2');

// El bot√≥n NO huye
function dodgeNo(){
  const stageRect = stage.getBoundingClientRect();
  const pad = 12;
  const maxX = stageRect.width - noBtn.offsetWidth - pad;
  const maxY = stageRect.height - noBtn.offsetHeight - pad;

  const x = pad + Math.random() * Math.max(0, maxX - pad);
  const y = pad + Math.random() * Math.max(0, maxY - pad);

  noBtn.style.left = x + 'px';
  noBtn.style.top  = y + 'px';
  noBtn.style.transform = 'none';
}

['pointerenter','pointermove','touchstart'].forEach(evt => {
  stage.addEventListener(evt, dodgeNo, { passive: true });
});

// YES ‚Üí segunda pantalla + confeti
yesBtn.addEventListener('click', () => {
  burstConfetti(120);
  screen1.classList.add("hidden");
  screen2.classList.remove("hidden");
});

function burstConfetti(n=80){
  for(let i=0;i<n;i++){
    const c = document.createElement('div');
    c.className = 'confetti';
    c.style.left = Math.random()*100 + 'vw';
    c.style.animationDuration = (1.4 + Math.random()*1.8) + 's';
    c.style.background = `hsl(${Math.floor(Math.random()*360)}, 90%, 60%)`;
    document.body.appendChild(c);
    setTimeout(()=>c.remove(), 3500);
  }
}

dodgeNo();
